// Funkcja pomocnicza AI Helper (ZMIENIONA NA PĘTLĘ RETRY)
async function handleAiHelperRequest(request, sendResponse) {
    const { prompt, apiKey, model, systemPrompt, fallbacks } = request;

    if (!apiKey) {
        sendResponse({ error: "Brak skonfigurowanego klucza API OpenRouter." });
        return;
    }

    const finalSystemPrompt = systemPrompt || "Jesteś profesjonalnym asystentem AI.";

    // 1. Budowanie kolejki modeli (Priorytet + Fallbacki)
    // Używamy Set, aby usunąć duplikaty (jeśli wybrany model jest też w fallbackach)
    let modelQueue = [model];
    if (fallbacks && Array.isArray(fallbacks)) {
        modelQueue = [...new Set([model, ...fallbacks])];
    }
    
    // Usuń ewentualne wartości "auto" lub puste z kolejki, jeśli się tam dostały
    modelQueue = modelQueue.filter(m => m && m !== 'auto');

    // Jeśli kolejka pusta, użyj auto-routera OpenRouter
    if (modelQueue.length === 0) modelQueue.push("openrouter/free");

    console.log("[Background] Rozpoczynam próbę z kolejką modeli:", modelQueue);

    // 2. Pętla Retry
    let lastError = null;
    for (const currentModel of modelQueue) {
        try {
            console.log(`[Background] Próba modelu: ${currentModel}...`);
            const messages = [
                { role: "system", content: finalSystemPrompt },
                { role: "user", content: prompt }
            ];

            const response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
                method: "POST",
                headers: {
                    "Authorization": `Bearer ${apiKey}`,
                    "Content-Type": "application/json",
                    "HTTP-Referer": "https://github.com/GeminiProPanel", 
                    "X-Title": "Gemini Pro Panel"
                },
                body: JSON.stringify({
                    model: currentModel, 
                    messages: messages,
                    // Opcjonalnie: mniejsza temperatura dla większej determinacji przy auto-zadaniach
                    temperature: 0.3 
                })
            });

            if (!response.ok) {
                const errText = await response.text();
                throw new Error(`HTTP ${response.status}: ${errText}`);
            }

            const data = await response.json();

            // Sprawdzenie czy jest treść (czasem API zwraca 200 ale puste choices przy błędach modelu)
            if (!data.choices || data.choices.length === 0 || !data.choices[0].message) {
                 throw new Error("Pusta odpowiedź od modelu (brak choices).");
            }

            const reply = data.choices[0].message.content || "";
            
            // SUKCES!
            // Zwracamy wynik i kończymy funkcję.
            console.log(`[Background] Sukces z modelem: ${currentModel}`);
            sendResponse({ result: reply, usedModel: currentModel });
            return;
        } catch (error) {
            console.warn(`[Background] Błąd modelu ${currentModel}:`, error.message);
            lastError = error.message;
            // Kontynuujemy pętlę do następnego modelu...
        }
    }

    // 3. Jeśli dotarliśmy tutaj, wszystkie modele zawiodły
    console.error("[Background] Wszystkie modele z puli zawiodły.");
    sendResponse({ error: `Wszystkie modele zawiodły. Ostatni błąd: ${lastError}` });
}

// --- HELPER: Znajdowanie pola input ---
function findInputField() {
  const selectors = [
    'div[contenteditable="true"][role="textbox"]',
    'div[contenteditable="true"]',
    'textarea[aria-label*="prompt"]', 
    'rich-textarea > div',
    'textarea'
  ];

  for (const sel of selectors) {
    const el = document.querySelector(sel);
    if (el) return el;
  }
  return null;
}

// --- NOWY HELPER: Pobieranie tekstu z inputu ---
function getTextInput(inputField) {
    if (!inputField) return "";

    // Preferuj innerText dla div[contenteditable] (Gemini)
    if (inputField.matches('div[contenteditable="true"]')) {
        return inputField.innerText;
    } 
    // Użyj value dla textarea
    else if (inputField.matches('textarea')) {
        return inputField.value;
    }
    // Fallback dla innych przypadków
    return inputField.textContent;
}

// --- LOGIKA TEKSTU ---
function injectTextToGemini(text, autoSubmit) {
  try {
    const inputField = findInputField();
    if (!inputField) return console.error("Błąd: Brak pola input.");

    inputField.focus();

    // Próba 1: execCommand
    const execSuccess = document.execCommand('insertText', false, text);

    // Próba 2: DOM Fallback
    if (!execSuccess) {
      inputField.innerText = text;
      const events = [
        new Event('input', { bubbles: true }),
        new KeyboardEvent('keydown', { key: 'Space', bubbles: true }),
        new KeyboardEvent('keyup', { key: 'Space', bubbles: true })
      ];
      events.forEach(ev => inputField.dispatchEvent(ev));
    }
    
    if (autoSubmit) handleAutoSubmit(600);
  } catch (err) {
    console.error("Błąd injectText:", err);
  }
}

// --- LOGIKA OBRAZU (NOWOŚĆ) ---
async function injectImageToGemini(base64Data, autoSubmit) {
  try {
    const inputField = findInputField();
    if (!inputField) return console.error("Błąd: Brak pola input dla obrazu.");

    inputField.focus();

    // 1. Konwersja Base64 -> Blob -> File
    const res = await fetch(base64Data);
    const blob = await res.blob();
    const file = new File([blob], "screenshot.png", { type: "image/png" });

    // 2. Tworzenie DataTransfer (Symulacja schowka)
    const dataTransfer = new DataTransfer();
    dataTransfer.items.add(file);

    // 3. Dispatch zdarzenia 'paste'
    // To jest kluczowe - Gemini nasłuchuje zdarzenia 'paste' dla plików
    const pasteEvent = new ClipboardEvent("paste", {
      bubbles: true,
      cancelable: true,
      clipboardData: dataTransfer
    });
    inputField.dispatchEvent(pasteEvent);
    console.log("[Gemini Panel] Wysłano zdarzenie paste z obrazkiem.");
    
    // Auto-Submit dla obrazków musi czekać dłużej (upload trwa)
    if (autoSubmit) handleAutoSubmit(2000);
  } catch (err) {
    console.error("Błąd injectImage:", err);
  }
}

function handleAutoSubmit(delay) {

        'button.send-button',

    let sendButton = null;

         const btn = document.querySelector(sel);
         if (btn && !btn.disabled && btn.getAttribute('aria-disabled') !== 'true') {
             sendButton = btn;
             break;
         }
    }

    if (sendButton) {
      sendButton.click();
      console.log("[Gemini Panel] Auto-Submit wykonany.");
    }
  }, delay);

/**
 * Funkcja wstrzykiwana do aktywnej karty w celu uruchomienia narzędzia do wycinania.
 * @returns {Promise<{x: number, y: number, width: number, height: number, dpr: number} | null>} Zwraca obszar wycięcia lub null, jeśli anulowano.
 */
export function injectCropTool() {
  // UWAGA: Ta funkcja jest WSTRZYKIWANA do kontekstu strony, a nie panelu.
  return new Promise((resolve) => {
    if (document.getElementById('gemini-crop-overlay')) return;

    const overlay = document.createElement('div');
    overlay.id = 'gemini-crop-overlay';
    overlay.style.cssText = `
      position: fixed; top: 0; left: 0; width: 100%; height: 100%;
      background: rgba(0, 0, 0, 0.4); z-index: 2147483647;
      cursor: crosshair; user-select: none;
    `;

    const selectionBox = document.createElement('div');
    selectionBox.style.cssText = `
      position: fixed; border: 2px solid #7c4dff; 
      background: rgba(124, 77, 255, 0.1); 
      box-shadow: 0 0 0 9999px rgba(0, 0, 0, 0.5);
      pointer-events: none; display: none;
    `;
    overlay.appendChild(selectionBox);
    document.body.appendChild(overlay);

    let startX, startY, isDragging = false;

    function onMouseDown(e) {
      if (e.button !== 0) return; // Tylko lewy przycisk
      isDragging = true;
      startX = e.clientX;
      startY = e.clientY;
      
      selectionBox.style.left = startX + 'px';
      selectionBox.style.top = startY + 'px';
      selectionBox.style.width = '0px';
      selectionBox.style.height = '0px';
      selectionBox.style.display = 'block';
    }

    function onMouseMove(e) {
      if (!isDragging) return;
      
      const currentX = e.clientX;
      const currentY = e.clientY;

      const width = Math.abs(currentX - startX);
      const height = Math.abs(currentY - startY);
      const left = Math.min(currentX, startX);
      const top = Math.min(currentY, startY);

      selectionBox.style.width = width + 'px';
      selectionBox.style.height = height + 'px';
      selectionBox.style.left = left + 'px';
      selectionBox.style.top = top + 'px';
    }

    function cleanup() {
      if (document.body.contains(overlay)) {
          document.body.removeChild(overlay);
      }
      window.removeEventListener('keydown', onKeyDown);
    }

    function onMouseUp(e) {
      if (!isDragging) return;
      isDragging = false;

      const rect = selectionBox.getBoundingClientRect();
      const dpr = window.devicePixelRatio || 1;

      const result = {
        x: rect.left,
        y: rect.top,
        width: rect.width,
        height: rect.height,
        dpr: dpr
      };
      
      // 1. Najpierw usuwamy nakładkę
      cleanup();
      
      // 2. Czekamy na "repaint" przeglądarki (kluczowe dla usunięcia fioletowej poświaty)
      // 100ms w zupełności wystarczy, by klatka się odświeżyła
      setTimeout(() => {
          if (rect.width < 5 || rect.height < 5) {
            resolve(null); // Zbyt mały obszar
          } else {
            resolve(result); // Dopiero teraz pozwalamy zrobić screena
          }
      }, 150);
    }

    function onKeyDown(e) {
      if (e.key === 'Escape') {
        cleanup();
        resolve(null); 
      }
    }

    overlay.addEventListener('mousedown', onMouseDown);
    overlay.addEventListener('mousemove', onMouseMove);
    overlay.addEventListener('mouseup', onMouseUp);
    window.addEventListener('keydown', onKeyDown);
  });
}

/**
 * Dodaje efekt błysku migawki do karty.
 * Wstrzykuje biały, półprzezroczysty div i natychmiast go usuwa.
 * @param {number} tabId ID karty.
 */
export async function playShutterEffect(tabId) {
    const effectFunc = () => {
        const div = document.createElement('div');
        div.id = 'gemini-shutter';
        div.style.cssText = `
            position: fixed; 
            inset: 0; 
            background: white; 
            z-index: 2147483647; 
            opacity: 0.8;
            transition: opacity 0.15s;
        `;
        document.documentElement.appendChild(div);

        // Natychmiastowe przejście do opacity 0 w następnej klatce
        requestAnimationFrame(() => {
            div.style.opacity = '0';
        });

        // Usunięcie elementu po zakończeniu animacji
        setTimeout(() => {
            if (document.documentElement.contains(div)) {
                document.documentElement.removeChild(div);
            }
        }, 200); // Czas dłuższy niż transition dla bezpieczeństwa
    };

    try {
        await chrome.scripting.executeScript({
            target: { tabId: tabId },
            func: effectFunc
        });
    } catch (error) {
        // Ignorujemy błędy, np. na chrome:// pages, gdzie nie można wstrzyknąć skryptu
        console.warn("Shutter effect injection failed:", error);
    }
}


/**
 * Wykonuje wycięcie (crop) obrazu DataURL na podstawie podanego obszaru.
 * @param {string} dataUrl Pełny DataURL zrzutu ekranu.
 * @param {{x: number, y: number, width: number, height: number, dpr: number}} area Obszar wycięcia.
 * @returns {Promise<Blob>} Obraz Blob wyciętego obszaru.
 */
export function cropImage(dataUrl, area) {
  return new Promise((resolve, reject) => {
    const img = new Image();
    img.onload = () => {
      const canvas = document.createElement('canvas');
      const scale = area.dpr; // Użycie Device Pixel Ratio
      canvas.width = area.width * scale;
      canvas.height = area.height * scale;
      
      const ctx = canvas.getContext('2d');
      // DrawImage: (obraz, sx, sy, sWidth, sHeight, dx, dy, dWidth, dHeight)
      ctx.drawImage(
        img, 
        area.x * scale, area.y * scale, area.width * scale, area.height * scale, 
        0, 0, canvas.width, canvas.height
      );
      
      // Zapis do Blob
      canvas.toBlob(blob => {
        if (blob) {
            resolve(blob);
        } else {
            reject(new Error("Błąd konwersji Canvas do Blob."));
        }
      }, 'image/png');
    };
    img.onerror = reject;
    img.src = dataUrl;
  });
}

/**
 * Konwertuje Blob na DataURL (Base64).
 * @param {Blob} blob 
 * @returns {Promise<string>} DataURL.
 */
export function blobToBase64(blob) {
  return new Promise((resolve, _) => {
    const reader = new FileReader();
    reader.onloadend = () => resolve(reader.result);
    reader.readAsDataURL(blob);
  });
}

import * as StorageService from '../StorageService.js';
import * as UIManager from '../UIManager.js';
import { GEMS_KEY } from '../Config.js';

let currentGems = [];

/**
 * Inicjalizuje moduł Gemów.
 */
export async function init() {
    const result = await chrome.storage.local.get([GEMS_KEY]);
    currentGems = result[GEMS_KEY] || [];
    renderGemsList();
}

/**
 * Zwraca listę gemów.
 */
export function getGems() {
    return [...currentGems];
}

/**
 * Dodaje nowy Gem do listy.
 */
export async function addGem(name, url) {
    if (!name || !url) {
        UIManager.showToast("Nazwa i URL są wymagane.", true);
        return;
    }

    const newGem = {
        id: Date.now(),
        name: name.trim(),
        url: url.trim()
    };

    currentGems.push(newGem);
    await chrome.storage.local.set({ [GEMS_KEY]: currentGems });
    
    renderGemsList();
    UIManager.showToast(`Dodano Gema: ${newGem.name}`);
}

/**
 * Usuwa Gem o podanym ID.
 */
export async function deleteGem(id) {
    currentGems = currentGems.filter(gem => gem.id !== id);
    await chrome.storage.local.set({ [GEMS_KEY]: currentGems });
    renderGemsList();
    UIManager.showToast("Gem usunięty.");
}

/**
 * Otwiera Gem w iframe.
 */
export function openGem(gem) {
    const iframe = document.getElementById('gemini-frame');
    if (iframe) {
        iframe.src = gem.url;
        UIManager.showToast(`Otwieranie Gema: ${gem.name}`);
    }
}

/**
 * Renderuje listę Gemów do elementu #gems-list.
 */
export function renderGemsList() {
    const listElement = document.getElementById('gems-list');
    if (!listElement) return;

    listElement.innerHTML = '';

    if (currentGems.length === 0) {
        listElement.innerHTML = '<li style="padding:10px; color:var(--text-muted); text-align:center;">Brak zapisanych Gemów.</li>';
        return;
    }

    currentGems.forEach(gem => {
        const li = document.createElement('li');
        li.className = 'template-item'; // Użycie istniejącej klasy dla spójności stylu
        li.style.cursor = 'pointer';
        
        li.innerHTML = `
            <div style="flex-grow:1; overflow:hidden;">
                <strong style="display:block;">${gem.name}</strong>
                <span style="font-size:10px; color:var(--text-muted); white-space:nowrap; overflow:hidden; text-overflow:ellipsis; display:block;">${gem.url}</span>
            </div>
            <div class="item-actions">
                <button class="icon-action-btn delete-btn" title="Usuń Gem"><i class="fas fa-trash"></i></button>
            </div>
        `;

        // Kliknięcie w element (poza przyciskiem usuwania) otwiera Gem
        li.addEventListener('click', (e) => {
            if (!e.target.closest('.delete-btn')) {
                openGem(gem);
            }
        });

        // Obsługa usuwania
        const delBtn = li.querySelector('.delete-btn');
        delBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            if (window.confirm(`Czy na pewno usunąć Gema "${gem.name}"?`)) {
                UIManager.animateRemoval(li, () => deleteGem(gem.id));
            }
        });

        listElement.appendChild(li);
    });
}

/**
 * Zapisuje notatki do pamięci i opcjonalnie animuje najnowszy element w UI.
 */
async function saveNotesAndUI() {
    try {
        await StorageService.saveNotes(currentNotes);
        // Po zapisie odświeżamy listę, aby pokazać nowe tagi, jeśli się pojawiły
        renderFilteredNotes(); 
    } catch (err) {
        if (err.message && (err.message.includes('QUOTA') || err.message.includes('quota'))) {
             UIManager.showToast("Przekroczono limit pamięci notatnika (5MB).", true);
        } else {
             UIManager.showToast("Błąd zapisu. Spróbuj skrócić tekst.", true);
        }
    }
}

/**
 * Filtruje notatki na podstawie currentSearchQuery i renderuje wynik.
 * @param {boolean} [animateNewest=false] Czy animować najnowszy element.
 */
export function renderFilteredNotes(animateNewest = false) {
    const query = currentSearchQuery.toLowerCase().trim();
    
    const filteredNotes = currentNotes.filter(note => {
        // Jeśli zapytanie jest puste, pokazujemy wszystko
        if (query.length === 0) return true;
        
        // Sprawdzanie tytułu
        if (note.title && note.title.toLowerCase().includes(query)) return true;
        
        // Sprawdzanie treści
        if (note.content && note.content.toLowerCase().includes(query)) return true;
        
        // Sprawdzanie tagów
        if (note.tags && Array.isArray(note.tags)) {
            // Sprawdzamy, czy którykolwiek tag pasuje
            return note.tags.some(tag => tag.toLowerCase().includes(query));
        }
        
        return false;
    });

    UIManager.renderNotebookList(
        filteredNotes,
        'notebook-list',
        deleteNote, 
        copyNoteContent, 
        openNote, // Przekazujemy nową funkcję otwierania
        animateNewest 
    );
}

 */
export function handleSearch(event) {
    currentSearchQuery = event.target.value;
    renderFilteredNotes(false); // Nie animujemy przy wyszukiwaniu
}


/**
 * Obsługuje zapis nowej notatki z asynchronicznym auto-tagowaniem i auto-tytułowaniem.
 */
export function handleSaveNote() {
    const titleInput = document.getElementById('note-title');
    const contentInput = document.getElementById('note-content');
    
    let title = titleInput.value.trim();
    const content = contentInput.value.trim();
    
    if (!content) {
        UIManager.showToast("Treść notatki nie może być pusta.", true);
        return;
    }
    
    let isAutoTitleNeeded = false;
    const now = new Date();

    // 1. Logika ustawiania tymczasowego tytułu (ACTION 2)
    if (!title) {
        // Data/Godzina w formie bardziej przyjaznej dla sortowania/filtrowania
        const dateStr = now.toLocaleDateString();
        const timeStr = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
        title = `Notatka ${dateStr} ${timeStr}`; // Tymczasowy tytuł
        isAutoTitleNeeded = true;
    }
    
    // --- 2. UTWORZENIE NOTATKI (Z pustą listą tagów) ---
    const newNote = {
        id: 'note_' + Date.now(),
        title: title,
        content: content,
        date: now.toISOString(),
        tags: [], // Inicjalizacja pustymi tagami
        isPinned: false // NOWY STAN: Domyślnie nieprzypięta
    };
    
    currentNotes.unshift(newNote); 
    
    // 3. Natychmiastowa aktualizacja UI (z animacją)
    renderFilteredNotes(true); // Używamy renderFilteredNotes, aby animować
    
    // 4. WSTĘPNY ZAPIS (w tle)
    StorageService.saveNotes(currentNotes).catch(err => {
        if (err.message && (err.message.includes('QUOTA') || err.message.includes('quota'))) {
             UIManager.showToast("Przekroczono limit pamięci notatnika (5MB).", true);
        } else {
             UIManager.showToast("Błąd zapisu wstępnego.", true);
        }
    });

    titleInput.value = '';
    contentInput.value = '';
    UIManager.showToast("Zapisano Snapshot! Trwa autotagowanie/tytułowanie...");
    
    // --- 5. ASYNCHRONICZNE AUTO-TAGOWANIE I AUTO-TYTUŁOWANIE ---
    
    (async () => {
        try {
            const settings = await StorageService.getSettings();
            if (!settings.apiKey) return;

            // TERAZ WYSTARCZY TO (Background obsłuży 'auto' i resztę):
            const effectiveModel = settings.model; 

            // Ograniczamy treść do tagowania/tytułowania, aby nie przekroczyć kontekstu.
            const contentForProcessing = newNote.content.substring(0, 1000); 

            let shouldSaveAndRender = false;

            // 1. AUTO-TITLE
            if (isAutoTitleNeeded) {
                // Skróć treść do generowania tytułu (oszczędność tokenów)
                const contentForTitle = newNote.content.substring(0, 500);
                // Prompt zmieniony zgodnie z instrukcją
                const titlePrompt = "Jesteś redaktorem. Wygeneruj krótki, konkretny tytuł (max 5 słów) dla podanego tekstu. Nie używaj cudzysłowów ani słowa 'Tytuł'.";
                 
                // Wywołanie AI (GeminiBridge.askAiHelper wyśle teraz listę fallbacków automatycznie)
                const titleResp = await GeminiBridge.askAiHelper(contentForTitle, settings.apiKey, effectiveModel, titlePrompt);
                 
                if (titleResp.result) {
                    const generatedTitle = titleResp.result.replace(/["']/g, "").trim();
                    // Znajdź notatkę po id i zaktualizuj tytuł
                    const idx = currentNotes.findIndex(n => n.id === newNote.id);
                    if (idx !== -1) {
                         // Zaktualizuj tylko, jeśli tytuł nie został zmieniony ręcznie w międzyczasie
                        if (currentNotes[idx].title === newNote.title) {
                            currentNotes[idx].title = generatedTitle;
                            shouldSaveAndRender = true;
                        }
                    }
                }
            }


            // 2. AUTO-TAGOWANIE (Istniejący kod)
            const tagSystemPrompt = "Jesteś botem kategoryzującym. Dla podanego tekstu zwróć TYLKO 3-5 słów kluczowych oddzielonych przecinkami. Nie używaj numeracji, hashtagów, ani innych znaków specjalnych. Słowa kluczowe powinny być zwięzłe.";
            
            // Używamy contentForProcessing (max 1000 znaków) dla tagów
            const tagResponse = await GeminiBridge.askAiHelper(contentForProcessing, settings.apiKey, effectiveModel, tagSystemPrompt);
            
            if (!tagResponse.error && tagResponse.result) {
                // Parsowanie tagów
                const rawTags = tagResponse.result.replace(/[\n\r]+/g, ' ').trim();
                const tagsArray = rawTags.split(',')
                    .map(tag => tag.trim())
                    .filter(tag => tag.length > 0)
                    .slice(0, 5); // Ograniczenie do max 5 tagów

                // Aktualizacja tagów
                const noteIndex = currentNotes.findIndex(n => n.id === newNote.id);
                if (noteIndex !== -1) {
                    currentNotes[noteIndex].tags = tagsArray;
                    shouldSaveAndRender = true;
                }
            } else {
                console.warn("[NotebookManager] Błąd autotagowania:", tagResponse.error);
            }
            
            // 3. AKTUALIZACJA NOTATKI I PONOWNY ZAPIS (jeśli nastąpiła zmiana)
            if (shouldSaveAndRender) {
                 await StorageService.saveNotes(currentNotes);
                 // Odświeżenie UI (bez animacji wstawiania)
                 renderFilteredNotes(false);
                 UIManager.showToast("Autotagowanie/tytułowanie zakończone!");
            }

        } catch (err) {
            console.error("[NotebookManager] Krytyczny błąd podczas autotagowania/tytułowania:", err);
            // Ciche zakończenie, aby nie rzucać błędem krytycznym do użytkownika.
        }
    })();
}

/**
 * Zapisuje edytowaną notatkę.
 */
export async function saveEditedNote() {
    if (!activeNoteId) return;

    const noteIndex = currentNotes.findIndex(n => n.id === activeNoteId);
    if (noteIndex === -1) {
        UIManager.showToast("Błąd: Nie znaleziono aktywnej notatki do zapisu.", true);
        return;
    }

    const titleInput = document.getElementById('edit-note-title');
    const contentTextarea = document.getElementById('edit-note-content');

    const newTitle = titleInput.value.trim();
    const newContent = contentTextarea.value.trim();

    if (!newContent) {
        UIManager.showToast("Treść notatki nie może być pusta. Użyj przycisku Usuń, jeśli chcesz ją skasować.", true);
        return;
    }
    
    // Aktualizacja obiektu (ACTION 3: Weryfikacja: newTitle pobiera z '#edit-note-title' - OK)
    currentNotes[noteIndex].title = newTitle || "Bez tytułu";
    currentNotes[noteIndex].content = newContent;
    currentNotes[noteIndex].date = new Date().toISOString(); // Aktualizacja daty modyfikacji
    
    // Triggery w NotebookManager:
    // 1. Zapisz do storage
    await saveNotesAndUI(); 

    // 2. Wróć do widoku listy
    UIManager.toggleNotebookView('list');
    activeNoteId = null;

    // 3. Odśwież listę (już zrobione przez saveNotesAndUI)
    // 4. Toast
    UIManager.showToast("Zapisano zmiany!");
}

/**
 * Przełącza flagę przypięcia dla aktywnej notatki.
 */
export async function togglePin() {
    if (!activeNoteId) return;

    const noteIndex = currentNotes.findIndex(n => n.id === activeNoteId);
    if (noteIndex === -1) {
        UIManager.showToast("Błąd: Nie znaleziono notatki do przypięcia.", true);
        return;
    }
    
    const isCurrentlyPinned = currentNotes[noteIndex].isPinned || false;
    currentNotes[noteIndex].isPinned = !isCurrentlyPinned;

    // Zapisz i odśwież listę (sortowanie zadziała w UIManager)
    await saveNotesAndUI(); 
    
    // Odśwież UI edytora (stan pinezki)
    const pinBtn = document.getElementById('btn-toggle-pin');
    if (pinBtn) {
        if (!isCurrentlyPinned) {
            pinBtn.classList.add('active');
            pinBtn.title = "Odepnij notatkę";
            UIManager.showToast(`Przypięto: ${currentNotes[noteIndex].title}`);
        } else {
            pinBtn.classList.remove('active');
            pinBtn.title = "Przypnij notatkę";
            UIManager.showToast(`Odepnięto: ${currentNotes[noteIndex].title}`);
        }
    }
}

    }
}

// Helper do pobierania unikalnych modeli z Config
function getUniqueModels() {
    const allModels = [];
    Object.values(Config.MODEL_GROUPS).forEach(g => {
        if(g.models) allModels.push(...g.models);
    });
    const uniqueModels = [];
    const seenIds = new Set();
    allModels.forEach(m => {
        if(!seenIds.has(m.id)) {
            uniqueModels.push(m);
            seenIds.add(m.id);
        }
    });
    return uniqueModels;
}

// ACTION 5: Funkcja pomocnicza do płynnej animacji licznika
function animateValue(obj, start, end, duration) {
    if (start === end) {
        obj.textContent = end.toLocaleString();
        
        // Wykonaj logikę aktualizacji pierścienia natychmiast
        updateRingUI(end);
        return;
    }

    const startTimestamp = performance.now();
    
    // Funkcja Logarytmicznego Easing (na potrzeby tylko liczby, pierścień ma własną logikę)
    function easeOutQuad(t) {
        return 1 - Math.pow(1 - t, 2);
    }

    function step(timestamp) {
        const elapsed = timestamp - startTimestamp;
        const progress = Math.min(1, elapsed / duration);
        
        // Easing function applied to progress
        const easedProgress = easeOutQuad(progress); 

        // Current value based on eased progress
        const currentValue = Math.round(start + (end - start) * easedProgress);

        obj.textContent = currentValue.toLocaleString();
        
        // Aktualizacja pierścienia (dla płynnego ruchu pierścienia - używamy aktualnej wartości)
        updateRingUI(currentValue);

        if (progress < 1) {
            requestAnimationFrame(step);
        } else {
            // Zapewnienie, że wartość końcowa jest dokładnie osiągnięta
            obj.textContent = end.toLocaleString();
            
            // Ostateczna aktualizacja pierścienia i statusu po zakończeniu animacji
            updateRingUI(end, true);
        }
    }

    requestAnimationFrame(step);
}

/**
 * Aktualizuje stan wizualny pierścienia (logika nieliniowa i kolory).
 * @param {number} currentTokenCount Aktualna liczba tokenów.
 * @param {boolean} isFinal Stan końcowy animacji.
 */
function updateRingUI(currentTokenCount, isFinal = false) {
    const ringEl = document.getElementById('token-hud-ring');
    const statusTextEl = document.getElementById('token-status-area')?.querySelector('#token-status-text');
    
    if (!ringEl || !statusTextEl) return;

    // --- 1. OBLICZANIE WIZUALNEGO WYPEŁNIENIA (Logarytmiczne/Nieliniowe) ---
    const ratio = currentTokenCount / MAX_CONTEXT_TOKENS;
    
    // Nieliniowe skalowanie: ratio ^ 0.25
    // Sprawia, że małe wartości (np. 25k/1M) wyglądają na bardziej wypełnione (~40%),
    // dając użytkownikowi lepszy feedback wizualny.
    const visualRatio = Math.pow(ratio, 0.25);
    const visualPercent = Math.min(100, Math.max(0, visualRatio * 100));

    // --- 2. AKTUALIZACJA CSS (Wypełnienie) ---
    // Aktualizujemy tylko zmienną --percentage
    ringEl.style.setProperty('--percentage', `${visualPercent}%`);
    
    // WAŻNE: USUNIĘTO aktualizację data-label, ponieważ procentu już nie wyświetlamy.
    ringEl.setAttribute('data-label', ''); // Ustawiamy na puste, aby usunąć tekst

    // --- 3. AKTUALIZACJA KOLORU (Liniowe) ---
    // Klasy kolorów (warning/danger) bazują na REALNYM, liniowym użyciu (ratio).
    ringEl.classList.remove('warning', 'danger');

    if (ratio > 0.75) {
        ringEl.classList.add('danger'); // Czerwony
    } else if (ratio > 0.50) {
        ringEl.classList.add('warning'); // Pomarańczowy
    }
    
    // --- 4. AKTUALIZACJA STATUSU I CZYSZCZENIE KLAS (TYLKO w końcowym kroku) ---
    if (isFinal) {
        ringEl.classList.remove('calculating');
        
        if (ratio > 1.0) {
            statusTextEl.textContent = 'LIMIT PRZEKROCZONY!';
            ringEl.classList.add('danger');
        } else if (ratio > 0.75) {
            statusTextEl.textContent = 'Wysokie zużycie!';
        } else if (ratio > 0.50) {
            statusTextEl.textContent = 'Średnie zużycie';
        } else if (ratio > 0.01) { // Pokaż status, jeśli coś wybrano
            statusTextEl.textContent = 'W normie';
        } else {
            statusTextEl.textContent = 'Wybierz karty';
        }
    }
}

/**
 * Parsuje string z numerami kart i zakresami.
 * np. "1-3, 5" -> [1, 2, 3, 5]
 * @param {string} inputString 
 * @param {number} maxTabs Maksymalny dozwolony indeks (liczba dostępnych kart).
 * @returns {Array<number>} Posortowana lista unikalnych, poprawnych indeksów (1-based).
 */
function parseTabSelection(inputString, maxTabs) {
    if (!inputString) return [];
    
    const parts = inputString.split(',').map(s => s.trim()).filter(s => s.length > 0);
    let indexes = new Set();

    for (const part of parts) {
        if (part.includes('-')) {
            const [startStr, endStr] = part.split('-').map(s => s.trim());
            const start = parseInt(startStr);
            const end = parseInt(endStr);

            if (!isNaN(start) && !isNaN(end) && start > 0 && end > 0) {
                const lower = Math.min(start, end);
                const upper = Math.min(Math.max(start, end), maxTabs); // Ograniczenie do maxTabs
                
                for (let i = lower; i <= upper; i++) {
                    if (i <= maxTabs) indexes.add(i);
                }
            }
        } else {
            const index = parseInt(part);
            if (!isNaN(index) && index > 0 && index <= maxTabs) {
                indexes.add(index);
            }
        }
    }

    return Array.from(indexes).sort((a, b) => a - b);
}

/**
 * Przełącza wybór karty i aktualizuje pole input.
 * @param {number} index Indeks karty (1-based).
 */
export function toggleClusterSelection(index) {
    const input = document.getElementById('cluster-input');
    const selectedIndices = parseTabSelection(input.value, clusterAvailableTabs.length);

    if (selectedIndices.includes(index)) {
        const newIndices = selectedIndices.filter(i => i !== index);
        input.value = newIndices.join(', ');
    } else {
        selectedIndices.push(index);
        input.value = selectedIndices.sort((a, b) => a - b).join(', ');
    }
    
    // Uruchomienie aktualizacji statystyk po zmianie
    updateClusterStats(parseTabSelection(input.value, clusterAvailableTabs.length));
}

/**
 * Szacuje tokeny dla wybranych kart i aktualizuje UI.
 * @param {Array<number>} selectedIndices Lista 1-based indeksów.
 */
export async function updateClusterStats(selectedIndices) {
    // 1. Dodanie klasy "calculating" na start
    const ringEl = document.getElementById('token-hud-ring');
    if (ringEl) {
        ringEl.classList.add('calculating');
        ringEl.classList.remove('warning', 'danger'); // Reset wizualny
    }

    const usageCountEl = document.getElementById('token-usage-count');
    const statusTextEl = document.getElementById('token-status-area')?.querySelector('#token-status-text');
    
    // Wróć do stanu początkowego, jeśli brak kart do analizy
    if (clusterAvailableTabs.length === 0) {
         animateValue(usageCountEl, 
             parseInt(usageCountEl.textContent.replace(/,/g, '') || '0'), 
             0, 0); 
         return;
    }
    
    let totalTokens = 0;
    
    // Ustawienie stanu ładowania
    usageCountEl.textContent = '...';
    // Ring jest ustawiany przez updateRingUI w animateValue

    if (selectedIndices.length === 0) {
        animateValue(usageCountEl, 
             parseInt(usageCountEl.textContent.replace(/,/g, '') || '0'), 
             0, 400); // Animacja do 0
        if(statusTextEl) statusTextEl.textContent = 'Wybierz karty';
        if (ringEl) ringEl.classList.remove('calculating');
        return;
    }

    // Pobranie tokenów (używając cache)
    const tokenPromises = selectedIndices.map(async (idx) => {
        const tab = clusterAvailableTabs[idx - 1];
        if (!tab || TabService.isRestrictedUrl(tab.url)) return 0;

        if (tokenEstimateCache[tab.id] !== undefined) {
            return tokenEstimateCache[tab.id];
        }

        const tokens = await TabService.estimateTokenCount(tab.id);
        tokenEstimateCache[tab.id] = tokens;
        return tokens;
    });

    const tokenResults = await Promise.all(tokenPromises);
    totalTokens = tokenResults.reduce((sum, current) => sum + current, 0);

    // Szacowanie tokenów promptu użytkownika (heurystyka 1/4)
    const customPromptText = document.getElementById('cluster-custom-prompt')?.value || "";
    const promptTokens = Math.ceil(customPromptText.length / 4);
    totalTokens += promptTokens;
    
    const currentVal = parseInt(usageCountEl.textContent.replace(/,/g, '') || '0');
    
    // ACTION 5: Zastąpienie bezpośredniego przypisania animateValue
    animateValue(usageCountEl, currentVal, totalTokens, 800);
}

/**
 * Pobiera treść z wybranych kart, buduje prompt i wysyła do Gemini.
 * @param {object} [options={}] Opcje wywołania.
 * @param {boolean} [options.silent=false] Jeśli true, pomija window.confirm.
 * @param {boolean} [options.analyzeAll=false] Jeśli true, wymusza analizę wszystkich kart.
 */
export async function handleClusterAnalysis(options = {}) {
    if(isAnalysisInProgress) return;
    isAnalysisInProgress = true;
    
    const input = document.getElementById('cluster-input');
    const customPromptInput = document.getElementById('cluster-custom-prompt');
    const progressBar = document.getElementById('cluster-progress-bar');
    const progressFill = document.getElementById('progress-fill');
    const progressText = document.getElementById('progress-text');
    const ringEl = document.getElementById('token-hud-ring'); // Pobierz referencję do pierścienia

    // 1. Pokaż pasek postępu i ustaw stan początkowy
    if (progressBar) progressBar.classList.remove('hidden');
    if (progressFill) progressFill.style.width = '0%';
    if (progressText) progressText.textContent = '0/0';
    
    // Ustawienie stanu ładowania na pierścieniu
    if (ringEl) ringEl.classList.add('calculating'); 

    try {
        let selectedIndices = parseTabSelection(input.value, clusterAvailableTabs.length);
        
        let targetTabs = [];
        
        // Logika wyboru kart na podstawie inputu lub opcji 'analyzeAll'
        if (selectedIndices.length === 0 && (input.value.trim() === "" || options.analyzeAll)) {
            // Puste pole = wszystkie
            targetTabs = clusterAvailableTabs;
            // W trybie analyzeAll nie jest to błąd
        } else if (selectedIndices.length === 0) {
            UIManager.showToast("Nie wybrano poprawnych kart.", true);
            return;
        } else {
            targetTabs = selectedIndices.map(idx => clusterAvailableTabs[idx - 1]).filter(tab => tab);
        }

        if (targetTabs.length === 0) {
            UIManager.showToast("Nie znaleziono kart do analizy.", true);
            return;
        }
        
        // Szacowanie tokenów (ponownie, na wypadek zmian)
        const tabIndicesForStats = targetTabs.map(tab => clusterAvailableTabs.findIndex(t => t.id === tab.id) + 1);
        await updateClusterStats(tabIndicesForStats);
        
        const totalTokensEl = document.getElementById('token-usage-count');
        const totalTokens = parseInt(totalTokensEl.textContent.replace(/,/g, '') || '0');

        if (totalTokens > MAX_CONTEXT_TOKENS) {
            UIManager.showToast("Kontekst jest zbyt duży. Proszę zmniejszyć liczbę wybranych kart.", true);
            return;
        }
        
        // Zapytanie o potwierdzenie, jeśli analiza jest duża I NIE jest w trybie silent
        if (targetTabs.length > 10 && !options.silent) {
            if(!window.confirm(`Wybrano ${targetTabs.length} kart. To może zająć chwilę i jest duże (${totalTokens.toLocaleString()} tokenów). Kontynuować?`)) {
                return;
            }
        }

        // Jeśli wywołane przez menu kontekstowe, nie zamykaj modala
        if (!options.silent) {
            UIManager.showToast(`Rozpoczynam analizę ${targetTabs.length} kart...`);
            UIManager.toggleModal('cluster-modal', false); 
        } else {
            UIManager.showToast(`Uruchomiono szybką analizę ${targetTabs.length} kart...`);
        }
        
        // REFACTOR: Wdrożenie licznika postępu
        let completed = 0;
        const total = targetTabs.length;

        const promisesWithProgress = targetTabs.map(tab => {
            // Owiń każde wywołanie w logikę aktualizacji postępu
            return TabService.extractTabContent(tab.id).then(result => {
                completed++;
                
                // Aktualizacja UI postępu
                const percentage = (completed / total) * 100;
                if (progressFill) progressFill.style.width = `${percentage}%`;
                if (progressText) progressText.textContent = `${completed}/${total}`;
                
                return result;
            });
        });
        
        // Oczekiwanie na wszystkie operacje (z wbudowanym postępem)
        const results = await Promise.all(promisesWithProgress);
        
        // 1. Czytanie custom promptu
        const customPrompt = customPromptInput.value.trim();
        
        // 2. Budowanie promptu
        const defaultPrompt = `Jesteś zaawansowanym asystentem do syntezy informacji. Twoim zadaniem jest przeanalizowanie treści z wielu źródeł i stworzenie spójnego podsumowania/analizy. Odpowiedz w języku polskim.`;
        
        let finalInstruction;
        if (customPrompt) {
            finalInstruction = `[USER INSTRUCTION]: ${customPrompt}`;
        } else {
            finalInstruction = `${defaultPrompt}\n\n[USER INSTRUCTION]: Zrób syntezę informacji z poniższych źródeł. Podsumuj kluczowe wnioski, podobieństwa i różnice.`;
        }

        let combinedContext = `${finalInstruction}\n\n[CONTEXT DATA]:\n`;
        
        let successCount = 0;
        
        results.forEach((res, i) => {
            if (res && !res.error && res.text.length > 0) {
                combinedContext += `--- ŹRÓDŁO #${i+1} (${res.title}) ---\n`;
                combinedContext += `URL: ${res.url}\n`;
                combinedContext += `TREŚĆ:\n${res.text}\n\n`;
                successCount++;
            }
        });
        
        if (successCount === 0) {
            UIManager.showToast("Nie udało się pobrać treści z żadnej karty.", true);
            return;
        }
        
        combinedContext += `\n--- END CONTEXT ---`;


        // Wysyłka do Gemini
        const settings = await StorageService.getSettings();
        const sent = GeminiBridge.sendToGeminiIframe(combinedContext, settings.autoSubmit);
        
        if (sent) {
            UIManager.showToast("Dane wysłane do Gemini (Cluster).");
        } else {
            await navigator.clipboard.writeText(combinedContext);
            UIManager.showToast("Dane w schowku (Gemini niedostępne).");
        }
        
    } catch (error) {
        console.error("Cluster Analysis Error:", error);
        UIManager.showToast("Błąd podczas analizy klastra.", true);
    } finally {
        isAnalysisInProgress = false;
        
        // 2. Ukryj pasek postępu i usuń stan ładowania z pierścienia
        if (progressBar) {
            // Daj chwilę na osiągnięcie 100% wizualnie
            setTimeout(() => {
                progressBar.classList.add('hidden');
                // Usunięcie calculating przeniesione do animateValue, ale robimy fallback
                if (ringEl && ringEl.classList.contains('calculating')) {
                     ringEl.classList.remove('calculating');
                     updateRingUI(parseInt(document.getElementById('token-usage-count').textContent.replace(/,/g, '') || '0'), true);
                }
            }, 500); 
        } else {
             if (ringEl && ringEl.classList.contains('calculating')) {
                 ringEl.classList.remove('calculating');
                 updateRingUI(parseInt(document.getElementById('token-usage-count').textContent.replace(/,/g, '') || '0'), true);
             }
        }
    }
}

/**
 * Obsługuje zapis nowego/edytowanego szablonu.
 */
export function handleSaveTemplate() {
    const nameInput = document.getElementById('new-template-name');
    const contentInput = document.getElementById('new-template-content');
    
    const name = nameInput.value.trim();
    let content = contentInput.value.trim();

    if (!name || !content) {
        UIManager.showToast("Podaj nazwę i treść szablonu.", true);
        return;
    }

    if (!content.includes('{text}')) {
        content += "\n\n---\n{text}\n---\n";
    }

    if (editingTemplateId) {
        const index = currentTemplates.findIndex(t => t.id === editingTemplateId);
        if (index !== -1) {
            currentTemplates[index].name = name;
            currentTemplates[index].content = content;
            UIManager.showToast("Zaktualizowano szablon!");
        } else {
            UIManager.showToast("Błąd: Nie znaleziono edytowanego szablonu.", true);
        }
        cancelEdit();
    } else {
        const newId = 'custom_' + Date.now();
        currentTemplates.push({ id: newId, name, content });
        
        nameInput.value = '';
        contentInput.value = '';
        UIManager.showToast("Dodano nowy szablon!");
    }

    saveTemplatesAndUI();
}

/**
 * Zamienia zmienne w treści szablonu na aktualne dane.
 * @param {string} templateContent Treść szablonu.
 * @param {{text: string, url: string, title: string, lang: string}} data Dane kontekstowe.
 */
export function processTemplateVariables(templateContent, data) {
    let processed = templateContent;
    
    processed = processed.replace(/{text}/g, data.text || '');
    processed = processed.replace(/{url}/g, data.url || '');
    processed = processed.replace(/{title}/g, data.title || '');
    
    const now = new Date();
    const dateStr = now.toISOString().split('T')[0];
    processed = processed.replace(/{date}/g, dateStr);
    
    processed = processed.replace(/{lang}/g, data.lang || 'nieznany');

    return processed;
}

 * @param {object} contextData Dane kontekstowe z karty.
 * @param {string|null} overrideTemplateId Opcjonalne ID szablonu do użycia jednorazowo (pomija selectedTemplateId).

export function getFinalPrompt(contextData, overrideTemplateId = null) {

/**
 * Wysyła zapytanie do AI Helper (Prompt Architect) przez background.js.
 * @param {string} prompt Szkic promptu do ulepszenia.
 * @param {string} apiKey Klucz API OpenRouter.
 * @param {string} model ID modelu.
 * @param {string | null} systemPrompt Opcjonalny prompt systemowy.
 * @returns {Promise<{result: string | undefined, error: string | undefined}>}
 */
export async function askAiHelper(prompt, apiKey, model, systemPrompt = null) {
    try {
        const response = await chrome.runtime.sendMessage({
            type: 'ask-ai-helper',
            prompt: prompt,
            apiKey: apiKey,
            model: model,       // Preferowany model
            
            fallbacks: FALLBACK_POOL, // <--- NOWE: Lista ratunkowa
            systemPrompt: systemPrompt
        });

        if (response && response.error) {
            // Rzuć błąd, aby pętla retry mogła go złapać
            throw new Error(response.error);
        }

        return response;
        // Zawiera { result: string, usedModel: string }
    } catch (err) {
        // Przechwytujemy błąd komunikacji (np. Chrome nie może wysłać wiadomości)
        console.error("Błąd komunikacji z background.js:", err);
        return { error: err.message || "Błąd komunikacji z AI Helper." };
    }
}

/**
 * Sprawdza, czy URL należy do zastrzeżonych (chrome://, extension://, itp.).
 * @param {string} url 
 * @returns {boolean}
 */
export function isRestrictedUrl(url) {
  if (!url) return true; // Bez URL nie mamy nic do roboty
  return url.startsWith('chrome://') || 
         url.startsWith('edge://') || 
         url.startsWith('about:') || 
         url.startsWith('chrome-extension://') ||
         url.includes('chrome.google.com/webstore') ||
         url.includes('gemini.google.com'); // Nie analizujemy samego Gemini
}

/**
 * Wstrzykiwana funkcja do rekurencyjnego trawersowania DOM.
 * Obsługuje Shadow DOM, iframy, elementy formularzy i formatowanie blokowe.
 * Ignoruje elementy niewidoczne i skrypty/style.
 * * ZAKTUALIZOWANA WERSJA: Dodano ekstrakcję atrybutów aria-label, title, placeholder (Deep Scrape).
 */
function getDeepTextContent(rootNode) {
    if (!rootNode) return "";
    
    // Lista tagów do całkowitego zignorowania
    const IGNORED_TAGS = ['SCRIPT', 'STYLE', 'NOSCRIPT', 'SVG', 'PATH', 'HEAD', 'META', 'LINK'];
    // Tagi blokowe
    const BLOCK_TAGS = ['P', 'DIV', 'H1', 'H2', 'H3', 'H4', 'H5', 'H6', 'LI', 'BR', 'UL', 'OL', 'TR', 'SECTION', 'ARTICLE', 'HEADER', 'FOOTER', 'NAV', 'BLOCKQUOTE'];
    
    let text = "";

    // Helper do widoczności
    const isVisible = (node) => {
        try {
            // Szybkie sprawdzenie dla node typu ELEMENT
            if (node.nodeType === Node.ELEMENT_NODE) {
                // Sprawdzamy style inline oraz atrybut hidden
                if (node.style && (node.style.display === 'none' || node.style.visibility === 'hidden')) return false;
                if (node.hasAttribute('hidden')) return false;
                if (node.getAttribute('aria-hidden') === 'true') return false;
                
                // Uproszczone sprawdzanie computed style (dla elementów, które nie są rootNode)
                if(node.getBoundingClientRect && (node.getBoundingClientRect().width === 0 || node.getBoundingClientRect().height === 0)) return false;
            }
            return true;
        } catch(e) { return true; }
    };

    if (!isVisible(rootNode)) return "";

    // Pobierz atrybuty semantyczne dla obecnego węzła (jeśli to Element)
    if (rootNode.nodeType === Node.ELEMENT_NODE) {
          const ariaLabel = rootNode.getAttribute('aria-label');
          const title = rootNode.getAttribute('title');
          // Dodajemy atrybut, jeśli istnieje, aby wzbogacić kontekst
          if (ariaLabel) text += ` [${ariaLabel}] `;
          else if (title) text += ` [${title}] `;
    }

    if (!rootNode.childNodes) return text;

    Array.from(rootNode.childNodes).forEach(node => {
        if (IGNORED_TAGS.includes(node.nodeName)) return;
        if (!isVisible(node)) return;

        // 1. TEXT NODE
        if (node.nodeType === Node.TEXT_NODE) {
            // Normalizacja białych znaków: zamienia wszystkie grupy spacji/nowych linii/tabulacji na pojedynczą spację
            const content = node.textContent.replace(/[\s\n\t]+/g, ' ').trim();
            if (content.length > 0) text += content + " ";
        }
        
        // 2. ELEMENT NODE
        else if (node.nodeType === Node.ELEMENT_NODE) {
            const tagName = node.tagName;
            let elementContent = "";

            // INPUT / TEXTAREA
            if (tagName === 'INPUT' || tagName === 'TEXTAREA') {
                const val = node.value;
                const placeholder = node.getAttribute('placeholder');
                // Preferujemy wartość pola, jeśli jest (np. dla formularzy)
                if (val && val.trim().length > 0) {
                    elementContent += ` [Input: ${val}] `;
                } 
                // Używamy placeholder'a jako kontekstu, jeśli pole jest puste
                else if (placeholder) {
                    elementContent += ` [Input: ${placeholder}] `;
                }
            }
            // IMG
            else if (tagName === 'IMG') {
                const alt = node.getAttribute('alt');
                if (alt) elementContent += ` [Img: ${alt}] `;
            }
            
            // Shadow DOM
            if (node.shadowRoot) {
                elementContent += getDeepTextContent(node.shadowRoot);
            }
            
            // IFRAME
            if (tagName === 'IFRAME') {
                try {
                    const doc = node.contentDocument || node.contentWindow.document;
                    if (doc && doc.body) elementContent += getDeepTextContent(doc.body);
                } catch (e) { 
                    // CORS block - dodajemy informację o bloku zamiast rzucać błędem
                    elementContent += "[IFRAME: Treść zablokowana przez CORS]"; 
                }
            }
            
            // Rekursja
            elementContent += getDeepTextContent(node);
            
            // Formatowanie
            if (BLOCK_TAGS.includes(tagName)) {
                // Dodajemy nową linię przed i po, jeśli treść jest znacząca
                if (elementContent.trim().length > 0) {
                     text += "\n" + elementContent.trim() + "\n";
                }
            } else {
                text += elementContent;
            }
        }
    });

    return text;
}

/**
 * Wykonuje skrypt w aktywnej karcie w celu pobrania jej treści.
 * Używa zaktualizowanej, rekurencyjnej funkcji.
 * @param {number} tabId ID karty.
 * @returns {Promise<{title: string, url: string, text: string, lang: string} | null>} Kontekst strony.
 */
export async function extractTabContent(tabId) {
    try {
        const result = await chrome.scripting.executeScript({
            target: { tabId: tabId },
            // Zmienione: Funkcja getDeepTextContent jest wstrzykiwana bezpośrednio, bez args,
            // dlatego musi być dostępna w kontekście background/service worker.
            func: getDeepTextContent, 
            args: [document.body] // Argumentem jest document.body
        });
        
        if (result && result[0] && result[0].result !== undefined) {
            let bodyText = result[0].result || "";
            
            // FINALNA LOGIKA FALLBACKU (Deep Scrape + Fallback)
            // Używamy bodyText jako 'text' z Deep Scrape
            let text = bodyText;
            
            // Normalizacja nadmiarowych białych znaków
            text = text.replace(/[\n\s]+/g, ' ').trim(); 

            // Jeśli wynik Deep Scrape jest podejrzanie krótki, użyjemy innerText
            if (!text || text.length < 100) {
                 const fallbackResult = await chrome.scripting.executeScript({
                    target: { tabId: tabId },
                    func: () => document.body.innerText
                 });
                 
                 const fallbackText = fallbackResult && fallbackResult[0] && fallbackResult[0].result ? fallbackResult[0].result : '';
                 
                 // Używamy fallbacku, jeśli jest dłuższy niż wynik Deep Scrape
                 if (fallbackText && fallbackText.length > text.length) {
                     text = fallbackText.replace(/[\n\s]+/g, ' ').trim(); // Normalizujemy fallback
                 }
            }
            
            // Ponowne czyszczenie po potencjalnym użyciu innerText (dla pewności)
            text = text.replace(/ \n/g, '\n'); 

            const tab = await chrome.tabs.get(tabId);
            const lang = tab.favIconUrl && tab.favIconUrl.includes('google.com') ? 'pl' : (tab.url && tab.url.includes('.pl') ? 'pl' : (document.documentElement ? document.documentElement.lang : 'en'));
            
            // Dodajemy limit znaków (choć ClusterManager ma własną logikę tokenów)
            const limit = CHAR_LIMIT;
            const isTruncated = text.length > limit;

            if (isTruncated) {
                text = text.substring(0, limit) + `\n\n[...ucięto ze względu na limit pamięci (${limit.toLocaleString()} znaków)]`;
            }

            return {
                title: tab.title,
                url: tab.url,
                text: text,
                lang: lang
            };
        }
        return null;
    } catch (err) {
        console.error(`Błąd pobierania treści z karty ${tabId}:`, err);
        return { error: true, tabId: tabId, text: '', url: '' }; // Zwracamy obiekt błędu
    }
}

/**
 * Wczytuje kontekst strony lub zaznaczenia dla aktywnej karty.
 * Używa zaktualizowanej, rekurencyjnej funkcji.
 * @returns {Promise<{text: string, url: string, title: string, lang: string, isSelection: boolean} | null>}
 */
export async function getPageContext() {
    try {
        const [tab] = await chrome.tabs.query({ active: true, lastFocusedWindow: true });
        
        if (!tab || isRestrictedUrl(tab.url)) {
            return { error: "Brak dostępu do strony.", tab: tab };
        }

        const result = await chrome.scripting.executeScript({
            target: { tabId: tab.id },
            func: () => { // Zmienione: Funkcja jest teraz self-contained (nie przyjmuje już deepTextContentFunc)
                // --- DEFINICJA HELPERA WEWNĄTRZ SCOPE'U STRONY ---
                function getDeepTextContent(rootNode) {
                    if (!rootNode) return "";
                    const IGNORED_TAGS = ['SCRIPT', 'STYLE', 'NOSCRIPT', 'SVG', 'PATH', 'HEAD', 'META', 'LINK'];
                    const BLOCK_TAGS = ['P', 'DIV', 'H1', 'H2', 'H3', 'H4', 'H5', 'H6', 'LI', 'BR', 'UL', 'OL', 'TR', 'SECTION', 'ARTICLE', 'HEADER', 'FOOTER', 'NAV', 'BLOCKQUOTE'];
                    
                    let text = "";
                    const isVisible = (node) => {
                        try {
                            if (node.nodeType === Node.ELEMENT_NODE) {
                                if (node.style && (node.style.display === 'none' || node.style.visibility === 'hidden')) return false;
                                if (node.hasAttribute('hidden')) return false;
                                if (node.getAttribute('aria-hidden') === 'true') return false;
                                if(node.getBoundingClientRect && (node.getBoundingClientRect().width === 0 || node.getBoundingClientRect().height === 0)) return false;
                            }
                            return true;
                        } catch(e) { return true; }
                    };

                    if (!isVisible(rootNode)) return "";

                    if (rootNode.nodeType === Node.ELEMENT_NODE) {
                          const ariaLabel = rootNode.getAttribute('aria-label');
                          const title = rootNode.getAttribute('title');
                          if (ariaLabel) text += ` [${ariaLabel}] `;
                          else if (title) text += ` [${title}] `;
                    }

                    if (!rootNode.childNodes) return text;

                    Array.from(rootNode.childNodes).forEach(node => {
                        if (IGNORED_TAGS.includes(node.nodeName)) return;
                        if (!isVisible(node)) return;

                        if (node.nodeType === Node.TEXT_NODE) {
                            const content = node.textContent.replace(/[\s\n\t]+/g, ' ').trim();
                            if (content.length > 0) text += content + " ";
                        }
                        else if (node.nodeType === Node.ELEMENT_NODE) {
                            const tagName = node.tagName;
                            let elementContent = "";

                            if (tagName === 'INPUT' || tagName === 'TEXTAREA') {
                                const val = node.value;
                                const placeholder = node.getAttribute('placeholder');
                                if (val && val.trim().length > 0) elementContent += ` [Input: ${val}] `;
                                else if (placeholder) elementContent += ` [Input: ${placeholder}] `;
                            }
                            else if (tagName === 'IMG') {
                                const alt = node.getAttribute('alt');
                                if (alt) elementContent += ` [Img: ${alt}] `;
                            }
                            
                            if (node.shadowRoot) elementContent += getDeepTextContent(node.shadowRoot);
                            
                            if (tagName === 'IFRAME') {
                                try {
                                    const doc = node.contentDocument || node.contentWindow.document;
                                    if (doc && doc.body) elementContent += getDeepTextContent(doc.body);
                                } catch (e) { elementContent += "[IFRAME: Treść zablokowana przez CORS]"; }
                            }
                            
                            elementContent += getDeepTextContent(node);
                            
                            if (BLOCK_TAGS.includes(tagName)) {
                                if (elementContent.trim().length > 0) {
                                     text += "\n" + elementContent.trim() + "\n";
                                }
                            } else {
                                text += elementContent;
                            }
                        }
                    });
                    return text;
                }
                // --- KONIEC HELPERA ---

                // --- GŁÓWNA LOGIKA ---
                const selection = window.getSelection().toString();
                const isSelection = (selection && selection.trim().length > 0);
                
                let text;
                
                if (isSelection) {
                    text = selection;
                } else {
                    // PRIMARY METHOD: Deep Scrape
                    text = getDeepTextContent(document.body);
                    
                    // FALLBACK METHOD: Native innerText (jeśli Deep Scrape zawiódł/pusty)
                    if (!text || text.trim().length < 100) {
                        const fallbackText = document.body.innerText;
                        // Użyj fallbacku, jeśli jest dłuższy i niepusty
                        if (fallbackText && fallbackText.length > (text ? text.length : 0)) {
                             text = fallbackText; 
                        }
                    }
                }
                
                return {
                    text: text,
                    isSelection: isSelection,
                    lang: document.documentElement.lang || navigator.language
                };
            },
            args: [] // Zmienione: brak argumentów
        });

        if (result && result[0] && result[0].result) {
            let { text, isSelection, lang } = result[0].result;
            
            // Jeśli użyliśmy Deep Scrape lub innerText (czyli nie było zaznaczenia), normalizujemy
            if (!isSelection) {
                 text = text.replace(/[\n\s]+/g, ' ').trim();
                 text = text.replace(/ \n/g, '\n');
            } else {
                // Jeśli jest zaznaczenie, po prostu trimujemy
                text = text.trim();
            }
            
            if (!text || text.length === 0) {
                 return { error: "Pusta strona lub zaznaczenie.", tab: tab };
            }

            const limit = CHAR_LIMIT;
            if (text.length > limit) {
                text = text.substring(0, limit) + `\n\n[...ucięto ze względu na limit pamięci (${limit.toLocaleString()} znaków)]`;
            }

            return {
                text: text,
                url: tab.url,
                title: tab.title,
                lang: lang,
                isSelection: isSelection
            };

        }
        return null;

    } catch (err) {
        console.error("Błąd pobierania kontekstu:", err);
        return { error: "Błąd skryptu Chrome.", tab: null };
    }
}

/**
 * Wyświetla menu kontekstowe w pozycji kursora.
 * @param {number} x Pozycja X myszy.
 * @param {number} y Pozycja Y myszy.
 * @param {Array<{label: string, id: string, isActive: boolean}>} items Lista opcji.
 * @param {Function} onSelect Callback wywoływany po wybraniu opcji (id).
 */
export function showContextMenu(x, y, items, onSelect) {
    // 1. Usuń istniejące menu (to wywoła też cleanup starych listenerów)
    closeContextMenu();

    // DODANO: Znajdź pasek narzędzi i dodaj klasę 'force-visible'
    const toolbar = document.querySelector('.toolbar');
    if (toolbar) {
        toolbar.classList.add('force-visible');
    }

    // 2. Stwórz element
    const menu = document.createElement('div');
    menu.className = 'custom-context-menu';
    menu.id = 'active-context-menu';
    
    const ul = document.createElement('ul');
    
    items.forEach(item => {
        const li = document.createElement('li');
        li.textContent = item.label;
        if (item.isActive) li.classList.add('active-template');
        
        // Dodaj ikonkę "Check" dla aktywnego (opcjonalnie)
        if (item.isActive) {
            li.innerHTML += ' <i class="fas fa-check" style="font-size: 10px;"></i>';
        }

        li.addEventListener('click', (e) => {
            e.stopPropagation();
            onSelect(item.id);
            closeContextMenu();
        });
        
        ul.appendChild(li);
    });

    menu.appendChild(ul);
    document.body.appendChild(menu);

    // 3. Pozycjonowanie (Detekcja krawędzi ekranu)
    const rect = menu.getBoundingClientRect();
    let finalX = x;
    let finalY = y;

    // Jeśli wychodzi poza prawą krawędź
    if (x + rect.width > window.innerWidth) {
        finalX = x - rect.width;
    }
    // Jeśli wychodzi poza dół
    if (y + rect.height > window.innerHeight) {
        finalY = y - rect.height;
    }

    menu.style.left = finalX + 'px';
    menu.style.top = finalY + 'px';

    // 4. Listener do zamykania przy kliknięciu poza (NAPRAWIONE)
    setTimeout(() => {
        // Definiujemy handler jako stałą referencję
        const handleOutsideClick = () => closeContextMenu();

        // Rejestrujemy funkcję czyszczącą, aby móc ją wywołać ręcznie w closeContextMenu
        activeMenuCleanup = () => {
            document.removeEventListener('click', handleOutsideClick);
            document.removeEventListener('contextmenu', handleOutsideClick);
        };

        // Dodajemy listenery
        document.addEventListener('click', handleOutsideClick, { once: true });
        document.addEventListener('contextmenu', handleOutsideClick, { once: true });
    }, 0);
}

/**
 * Zamyka aktywne menu kontekstowe.
 */
export function closeContextMenu() {
    // 1. Wykonaj cleanup listenerów dokumentu, jeśli istnieją
    if (activeMenuCleanup) {
        activeMenuCleanup();
        activeMenuCleanup = null;
    }

    // DODANO: Znajdź pasek narzędzi i usuń klasę 'force-visible'
    const toolbar = document.querySelector('.toolbar');
    if (toolbar) {
        toolbar.classList.remove('force-visible');
    }

    const existing = document.getElementById('active-context-menu');
    if (existing) {
        existing.remove();
    }
}

 * @param {object} modelGroups 
 */
export function populateModelSelect(selectElement, modelGroups = MODEL_GROUPS) {

/**
 * Eksportowana funkcja do animowania usuwania elementu listy (Smooth Collapse).
 * Dodaje klasę collapse, czeka na koniec CSS transition, a następnie wykonuje callback.
 * @param {HTMLElement} element Element li do animacji.
 * @param {Function} callback Funkcja do wywołania po zakończeniu animacji (np. usunięcie danych).
 */
export function animateRemoval(element, callback) {
    if (!element) {
        callback();
        return;
    }

    // Ustawienie początkowej wysokości na rzeczywistą wartość dla płynnego przejścia
    element.style.height = element.offsetHeight + 'px';
    
    // Wymuszenie reflow, aby przeglądarka zastosowała wysokość przed dodaniem klasy .list-item-collapsing
    element.offsetHeight; 

    // Dodanie klasy triggerującej animację (ustawia height: 0, opacity: 0)
    element.classList.add('list-item-collapsing');

    // Czas trwania animacji zdefiniowany w CSS: --collapse-duration (0.35s)
    const transitionDuration = 350; 

    setTimeout(() => {
        // Po zakończeniu animacji, wykonaj faktyczną logikę usunięcia/przeładowania listy
        callback(); 
    }, transitionDuration);
}

/**
 * Eksportowana funkcja symulująca efekt pisania na maszynie.
 * @param {HTMLTextAreaElement | HTMLInputElement} element Element do wpisywania tekstu.
 * @param {string} text Pełny tekst do wpisania.
 * @param {number} speed Opóźnienie między znakami w ms.
 */
export function typeWriterEffect(element, text, speed = 15) {
    let i = 0;
    
    // Wyczyść i zablokuj pole
    element.value = '';
    element.readOnly = true;
    
    // ACTION 4: Dodaj klasę wizualną aktywnego pisania
    element.classList.add('typing-active');

    function type() {
        if (i < text.length) {
            element.value += text.charAt(i);
            // Auto-scroll w dół
            element.scrollTop = element.scrollHeight;
            i++;
            
            // ACTION 4: Dodaj losowość do speed
            const randomSpeed = speed + Math.random() * 10;
            
            setTimeout(type, randomSpeed);
        } else {
            // Po zakończeniu, odblokuj pole
            element.readOnly = false;
            // ACTION 4: Usuń klasę wizualną
            element.classList.remove('typing-active');
        }
    }
    
    type();
}

/**

 * @param {string} containerId ID elementu UL/OL/DIV, w którym ma się pojawić loader.
 * @param {number} count Liczba elementów szkieletu do wygenerowania.

    const container = document.getElementById(containerId);
    if (!container) return;

    // Usuń bieżącą zawartość
    container.innerHTML = '';
    container.classList.add('skeleton-loading');

    // Generuj elementy szkieletu
    for (let i = 0; i < count; i++) {
        const skeletonItem = document.createElement('li'); // Użyj li, zakładając listy
        skeletonItem.className = 'skeleton-item';
        
        // Dodaj placeholder dla treści (np. dwa divy symulujące tytuł i podtytuł)
        skeletonItem.innerHTML = `
            <div style="width: 70%; height: 10px; background-color: rgba(255, 255, 255, 0.05); margin-bottom: 5px; border-radius: 4px;"></div>
            <div style="width: 90%; height: 8px; background-color: rgba(255, 255, 255, 0.03); border-radius: 4px;"></div>
        `;
        
        // Używamy setTimeout, aby uniknąć zablokowania UI podczas szybkiego odświeżania.
        setTimeout(() => {
            container.appendChild(skeletonItem);
        }, 0);
    }
}


/**
 * Renderuje listę szablonów w UI.
 * @param {Array<Object>} templates 
 * @param {string} containerId 
 * @param {Function} onDelete 
 * @param {Function} onEdit 
 * @param {Array<Object>} defaultTemplates Lista szablonów domyślnych do sprawdzenia, czy można usunąć.
 */
export function renderTemplatesList(templates, containerId, onDelete, onEdit, defaultTemplates) {
    const list = document.getElementById(containerId);
    if (!list) return;
    
    // Usunięcie klasy loading, jeśli była
    list.classList.remove('skeleton-loading');
    list.innerHTML = '';

    templates.forEach(tpl => {
        const isDefault = defaultTemplates.some(dt => dt.id === tpl.id);
        
        const li = document.createElement('li');
        li.className = 'template-item';
        
        li.innerHTML = `
        <div style="flex-grow:1; margin-right: 15px; overflow: hidden;">
            <strong style="display:block; margin-bottom: 2px;">${tpl.name}</strong>
            <span style="font-size:11px; color:var(--text-muted); display:block; text-overflow: ellipsis; white-space: nowrap; overflow: hidden;">
            ${tpl.content.substring(0, 60)}${tpl.content.length > 60 ? '...' : ''}
            </span>
        </div>
        <div class="item-actions">
            <button class="icon-action-btn edit-btn" data-id="${tpl.id}" title="Edytuj"><i class="fas fa-edit"></i></button>
            ${!isDefault ? `<button class="icon-action-btn delete-btn" data-id="${tpl.id}" title="Usuń"><i class="fas fa-trash"></i></button>` : ''}
        </div>
        `;
        list.appendChild(li);

        li.querySelector('.edit-btn').addEventListener('click', (e) => {
            e.stopPropagation();
            onEdit(tpl.id);
        });

        if (!isDefault) {
            const delBtn = li.querySelector('.delete-btn');
            delBtn.addEventListener('click', (e) => {
                e.stopPropagation();
                // REFACTOR: Użycie animateRemoval przed usunięciem
                const confirmDelete = window.confirm("Czy na pewno usunąć ten szablon?");
                if (confirmDelete) {

        empty.style.padding = '20px';
        empty.style.textAlign = 'center';
        empty.style.color = 'var(--text-muted)';
        empty.innerText = 'Brak zapisanych szablonów.';
        list.appendChild(empty);
    }
}

/**
 * Parsuje podstawowy Markdown na HTML.
 * Obsługiwane: **pogrubienie**, *kursywa*, # Nagłówek, - lista, \n nowa linia.
 * @param {string} text Tekst z Markdown.
 * @returns {string} Wyrenderowany HTML.
 */
export function simpleMarkdownParser(text) {
    if (!text) return '';

    // 1. Zamień znaki nowej linii na placeholder
    let html = text.replace(/\n/g, '__BR_PLACEHOLDER__');

    // 2. Nagłówki (tylko # - użyjemy <h3>)
    // Dopasowuje linię zaczynającą się od #, po której jest spacja, aż do końca linii
    html = html.replace(/^#\s*(.*)$/gm, '<h3>$1</h3>');

    // 3. Pogrubienie: **tekst** na <b>tekst</b>
    html = html.replace(/\*\*(.*?)\*\*/g, '<b>$1</b>');

    // 4. Kursywa: *tekst* na <i>tekst</i>
    html = html.replace(/\*(.*?)\*/g, '<i>$1</i>');
    
    // 5. Listy: - item (owinięte w <ul>), usuwa __BR_PLACEHOLDER__ w listach
    // A. Znajdź wszystkie bloki listy
    let listBlocks = [];
    let currentList = [];
    let inList = false;

    // Podziel na linie, aby obsłużyć bloki listy
    const lines = html.split('__BR_PLACEHOLDER__');
    
    lines.forEach(line => {
        if (line.trim().startsWith('- ')) {
            // Dodaj item do bieżącej listy
            currentList.push(`<li>${line.trim().substring(2).trim()}</li>`);
            inList = true;
        } else {
            // Jeśli byliśmy w liście i natrafiliśmy na nie-listę, zakończ blok
            if (inList) {
                listBlocks.push(`<ul>${currentList.join('')}</ul>`);
                currentList = [];
                inList = false;
            }
            // Zwykła linia: dodaj ją z powrotem jako placeholder + treść
            if (line.trim().length > 0) {
                 listBlocks.push(line);
            }
        }
    });

    // Dodaj ostatni blok listy, jeśli istnieje
    if (inList) {
        listBlocks.push(`<ul>${currentList.join('')}</ul>`);
    }

    // Złóż z powrotem, używając <br> dla separatora
    html = listBlocks.join('<br>');
    
    // 6. Zamień pozostałe placeholdery na <br> (tylko te, które nie są w <ul>/<h3>)
    html = html.replace(/__BR_PLACEHOLDER__/g, '<br>');

    // 7. Upewnij się, że nie ma pustych br na początku/końcu
    html = html.replace(/^(<br\s*\/?>)+/, '').replace(/(<br\s*\/?>)+$/, '');
    
    // 8. Dodaj owinięcie w <p>, jeśli nie jest to blok (H3, UL) - UPRASZCZAMY: pomijamy dla czytelności.
    
    return html.trim();
}

/**
 * Renderuje listę notatek w notatniku.
 * Zmieniono sygnaturę: dodano onOpen (otwieranie edytora).
 * @param {Array<Object>} notes 
 * @param {string} containerId 
 * @param {Function} onDelete 
 * @param {Function} onCopy Callback na kliknięcie notatki (kopiowanie).
 * @param {Function} onOpen Callback na otwarcie notatki do edycji (lub onCopy jako fallback).
 * @param {boolean} [animateFirst=false] Czy animować pierwszy element (najnowszy).
 */
export function renderNotebookList(notes, containerId, onDelete, onCopy, onOpen, animateFirst = false) {
    const list = document.getElementById(containerId);
    if (!list) return;
    
    // Usunięcie klasy loading, jeśli była
    list.classList.remove('skeleton-loading');
    list.innerHTML = '';
    
    // SORTOWANIE (przypięte na górze, potem po dacie)
    notes.sort((a, b) => {
        // 1. Przypięte na górze
        if (a.isPinned && !b.isPinned) return -1;
        if (!a.isPinned && b.isPinned) return 1;

        // 2. Po dacie (najnowsze na górze)
        return new Date(b.date) - new Date(a.date);
    });

    if (notes.length === 0) {
        const empty = document.createElement('li');
        empty.style.padding = '20px';
        empty.style.textAlign = 'center';
        empty.style.color = 'var(--text-muted)';
        empty.innerText = 'Brak zapisanych snapshotów.';
        list.appendChild(empty);
        return;
    }

    notes.forEach((note, index) => {
        const li = document.createElement('li');
        
        // DODANO: Warunek na animację najnowszego elementu
        li.className = 'note-item';
        if (note.isPinned) {
            li.classList.add('pinned-note'); // Dodanie klasy dla przypiętych
        }
        if (index === 0 && animateFirst === true) {
            li.classList.add('list-item-new'); // Klasa triggerująca animację
        }
        
        const dateObj = new Date(note.date);
        const dateStr = dateObj.toLocaleDateString() + ' ' + dateObj.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
        
        // Generowanie HTML dla tagów
        let tagsHtml = '';
        if (note.tags && Array.isArray(note.tags) && note.tags.length > 0) {
            const tags = note.tags.map(tag => `<span class="note-tag">#${tag}</span>`).join('');
            tagsHtml = `<div class="note-tags">${tags}</div>`;
        }
        
        // Renderujemy PINEZKĘ w nagłówku notatki, jeśli jest przypięta
        const pinIcon = note.isPinned ? `<i class="fas fa-thumbtack" style="color:#FFD700; margin-right: 5px;"></i>` : '';

        li.innerHTML = `
            <div class="note-header">
                <span class="note-title">${pinIcon}${note.title}</span>
                <span class="note-date">${dateStr}</span>
            </div>
            <div class="note-preview">${note.content}</div>
            ${tagsHtml}
            <div class="item-actions" style="margin-top: 5px;">
                <button class="icon-action-btn edit-btn" data-id="${note.id}" title="Edytuj"><i class="fas fa-edit"></i></button>
                <button class="icon-action-btn copy-btn" data-id="${note.id}" title="Kopiuj"><i class="fas fa-copy"></i></button>
                <button class="icon-action-btn delete-btn" data-id="${note.id}" title="Usuń"><i class="fas fa-trash"></i></button>
            </div>
        `;
        
        // Listener na przycisk EDYCJI
        const editBtn = li.querySelector('.edit-btn');
        editBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            onOpen(note.id); // Otwórz edytor
        });

        // Listener na przycisk KOPIOWANIA
        const copyBtn = li.querySelector('.copy-btn');
        copyBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            onCopy(note.content); // Skopiuj treść
        });
        
        // Listener na przycisk usuwania
        const delBtn = li.querySelector('.delete-btn');
        delBtn.addEventListener('click', (e) => {
            e.stopPropagation(); 
            
            // REFACTOR: Użycie animateRemoval przed usunięciem
            const confirmDelete = window.confirm("Usunąć ten snapshot?");
            if (confirmDelete) {
                animateRemoval(li, () => onDelete(note.id));
            }
        });
        
        list.appendChild(li);
    });
}

/**
 * Przełącza widok modala notatnika.
 * @param {'list' | 'editor'} viewName Nazwa widoku do pokazania.
 */
export function toggleNotebookView(viewName) {
    const listView = document.getElementById('notebook-list-view');
    const editorView = document.getElementById('notebook-editor-view');
    const footerList = document.getElementById('notebook-footer-list');
    const footerEditor = document.getElementById('notebook-footer-editor');
    const modalContainer = document.getElementById('notebook-modal'); // ACTION: Pobierz kontener modala
    
    if (viewName === 'editor') {
        // ACTION: Dodaj klasę stanu edycji
        if (modalContainer) modalContainer.classList.add('view-editor');

        // Pokaż Edytor, Ukryj Listę
        listView.classList.add('hidden');
        editorView.classList.remove('hidden');
        footerList.classList.add('hidden');
        footerEditor.classList.remove('hidden');
        
        // Upewnij się, że edytor jest w stanie początkowym (tab Edycja)
        const editTabBtn = document.getElementById('btn-tab-edit');
        if (editTabBtn) editTabBtn.click(); 
    } else { // 'list'
        // ACTION: Usuń klasę stanu edycji
        if (modalContainer) modalContainer.classList.remove('view-editor');

        // Pokaż Listę, Ukryj Edytor
        listView.classList.remove('hidden');
        editorView.classList.add('hidden');
        footerList.classList.remove('hidden');
        footerEditor.classList.add('hidden');
    }
}

/**
 * Wypełnia edytor notatki danymi.
 * @param {Object} note Obiekt notatki.
 */
export function fillNoteEditor(note) {
    const titleInput = document.getElementById('edit-note-title');
    const contentTextarea = document.getElementById('edit-note-content');
    const previewDiv = document.getElementById('edit-note-preview');
    const pinBtn = document.getElementById('btn-toggle-pin');
    const lastModifiedSpan = document.getElementById('last-modified-date');
    
    // 1. Wypełnij pola
    titleInput.value = note.title;
    contentTextarea.value = note.content;
    
    // 2. Ustaw stan pinezki
    if (note.isPinned) {
        pinBtn.classList.add('active');
        pinBtn.title = "Odepnij notatkę";
    } else {
        pinBtn.classList.remove('active');
        pinBtn.title = "Przypnij notatkę";
    }
    
    // 3. Renderuj podgląd (opcjonalnie, ale dobrze jest mieć na start)
    // UWAGA: Robimy to tylko, jeśli jesteśmy w widoku 'preview' (co jest domyślnie ukryte)
    // Opcjonalnie: możemy wywołać renderowanie na pierwszym przełączeniu karty.
    previewDiv.innerHTML = simpleMarkdownParser(contentTextarea.value); // Użyj contentTextarea.value, a nie note.content, w razie wcześniejszej edycji
    
    // 4. Ustaw datę modyfikacji w stopce
    const dateObj = new Date(note.date);
    const dateStr = dateObj.toLocaleDateString() + ' ' + dateObj.toLocaleTimeString();
    lastModifiedSpan.textContent = dateStr;
}

/**
 * Inicjalizuje listenery dla tabulatorów edytora notatnika.
 */
export function setupEditorTabs(onPreviewRender) {
    const editTab = document.getElementById('btn-tab-edit');
    const previewTab = document.getElementById('btn-tab-preview');
    const contentTextarea = document.getElementById('edit-note-content');
    const previewDiv = document.getElementById('edit-note-preview');

    const setActiveTab = (activeBtn, inactiveBtn, activeElement, inactiveElement) => {
        activeBtn.classList.add('active');
        inactiveBtn.classList.remove('active');
        activeElement.classList.remove('hidden');
        inactiveElement.classList.add('hidden');
    };

    if (editTab && previewTab && contentTextarea && previewDiv) {
        editTab.addEventListener('click', () => {
            setActiveTab(editTab, previewTab, contentTextarea, previewDiv);
        });

        previewTab.addEventListener('click', () => {
            // Renderuj Markdown przed przełączeniem
            previewDiv.innerHTML = simpleMarkdownParser(contentTextarea.value);
            onPreviewRender(); // Callback do NotebookManagera, aby odświeżyć dane/editor-view

            setActiveTab(previewTab, editTab, previewDiv, contentTextarea);
        });
        
        // Dodatkowo: upewnij się, że podgląd się odświeża podczas pisania w trybie podglądu
        contentTextarea.addEventListener('input', () => {
            if (previewTab.classList.contains('active')) {
                previewDiv.innerHTML = simpleMarkdownParser(contentTextarea.value);
            }
        });
    }
}


/**
 * Wypełnia listę kart w modal Cluster.
 * @param {Array<chrome.tabs.Tab>} tabs 
 * @param {Function} onTabClick 
 */
export function renderClusterTabs(tabs, onTabClick) {
    const list = document.getElementById('cluster-tabs-list');
    if (!list) return;
    list.innerHTML = '';
        
    if (tabs.length === 0) {
        list.innerHTML = '<li style="padding:10px; color:var(--text-muted);">Brak dostępnych kart do analizy.</li>';

    }

    tabs.forEach((tab, index) => {
        const li = document.createElement('li');
        li.className = 'tab-list-item';
        const displayIndex = index + 1;
        
        li.innerHTML = `
            <span class="tab-badge">${displayIndex}</span>
            <div class="tab-info">
               <span class="tab-title">${tab.title || 'Bez tytułu'}</span>
               <span class="tab-url">${new URL(tab.url).hostname}</span>
            </div>
        `;
        
        li.addEventListener('click', () => {
            onTabClick(displayIndex);
        });
        
        list.appendChild(li);
    });
}

    icon.className = 'fas fa-circle-notch fa-spin';
    try {
        const [tab] = await chrome.tabs.query({ active: true, lastFocusedWindow: true });
        if (!tab || TabService.isRestrictedUrl(tab.url)) {
            UIManager.showToast("Nie można zrobić zrzutu na tej stronie.", true);
            icon.className = originalIconClass; 
            return;
        }
        
        const dataUrl = await CaptureService.captureVisibleTab();
        CaptureService.playShutterEffect(tab.id).catch(err => console.warn(err));

        let clipboardSuccess = false;
        try {
            const res = await fetch(dataUrl);
            const blob = await res.blob();
            await navigator.clipboard.write([
              new ClipboardItem({ [blob.type]: blob })
            ]);
            clipboardSuccess = true;
        } catch (clipboardErr) {
            console.warn("Clipboard warning:", clipboardErr);
        }
        
        const settings = await StorageService.getSettings();
        const sent = GeminiBridge.sendToGeminiIframeImage(dataUrl, settings.autoSubmit);

        icon.className = originalIconClass;

        if (sent) {
            const msg = settings.autoSubmit ?
            "Wysłano zrzut do Gemini!" : "Wklejono zrzut do Gemini!";
            UIManager.showToast(msg);
            UIManager.showButtonFeedback(btnId, true);
        } else if (clipboardSuccess) {
            UIManager.showToast("Zrzut ekranu w schowku.");
            UIManager.showButtonFeedback(btnId, true);
        } else {
             UIManager.showToast("Problem ze schowkiem, sprawdź czat.", true);
        }

    } catch (err) {
        console.error("Screenshot error:", err);
        UIManager.showToast("Błąd zrzutu ekranu.", true);
        
        icon.className = originalIconClass;
        UIManager.showButtonFeedback(btnId, false);
    }
}

// 2. WYCINEK EKRANU (Crop Tool)
async function handleCrop() {
    const btnId = 'btn-crop';
    try {
        const [tab] = await chrome.tabs.query({ active: true, lastFocusedWindow: true });
        if (!tab || TabService.isRestrictedUrl(tab.url)) {
            UIManager.showToast("Nie można zrobić zrzutu na tej stronie.", true);
            return;
        }

        // A. Wybór obszaru
        const injectionResults = await chrome.scripting.executeScript({
          target: { tabId: tab.id },
          func: CaptureService.injectCropTool
        });
        const cropArea = injectionResults[0].result;
        if (!cropArea) return; // Anulowano

        // B. Przetwarzanie - START SPINNERA
        const btn = document.getElementById(btnId);
        const icon = btn.querySelector('i');
        const originalIconClass = 'fas fa-crop-simple';
        icon.className = 'fas fa-circle-notch fa-spin';
        // C. Capture & Crop
        const fullScreenshotDataUrl = await CaptureService.captureVisibleTab();
        
        CaptureService.playShutterEffect(tab.id).catch(err => console.warn(err));
        const croppedBlob = await CaptureService.cropImage(fullScreenshotDataUrl, cropArea);
        
        // Backup: Schowek
        let clipboardSuccess = false;
        try {
            await navigator.clipboard.write([
               new ClipboardItem({ 'image/png': croppedBlob })
            ]);
            clipboardSuccess = true;
        } catch (clipboardErr) {
            console.warn("Clipboard warning:", clipboardErr);
        }
        
        // D. Ghost Writer
        const base64Data = await CaptureService.blobToBase64(croppedBlob);
        const settings = await StorageService.getSettings();
        const sent = GeminiBridge.sendToGeminiIframeImage(base64Data, settings.autoSubmit);

        icon.className = originalIconClass;
        if (sent) {
            const msg = settings.autoSubmit ?
            "Wysłano wycinek do Gemini!" : "Wklejono wycinek do Gemini!";
            UIManager.showToast(msg);
            UIManager.showButtonFeedback(btnId, true);
        } else if (clipboardSuccess) {
            UIManager.showToast("Wycinek w schowku.");
            UIManager.showButtonFeedback(btnId, true);
        } else {
             UIManager.showToast("Problem ze schowkiem, sprawdź czat.", true);
        }

    } catch (err) {
        console.error("Crop error:", err);
        UIManager.showToast("Błąd wycinania ekranu.", true);
        
        const btn = document.getElementById(btnId);
        if(btn) {
            btn.querySelector('i').className = 'fas fa-crop-simple';
            UIManager.showButtonFeedback(btnId, false);
        }
    }
}


// 3. CONTEXT - Zbieranie danych + GHOST WRITER (Text)
async function handleContext(overrideTemplateId = null) {
    const btnId = 'btn-context';
    try {
        const contextData = await TabService.getPageContext();
        if (contextData && contextData.error) {
            UIManager.showToast(contextData.error, true);
            UIManager.showButtonFeedback(btnId, false);
            return;
        }

        if (!contextData) throw new Error("Nie udało się pobrać danych kontekstowych.");
        const finalPrompt = TemplateManager.getFinalPrompt(contextData, overrideTemplateId);
        
        let templateName;
        if (overrideTemplateId) {
            const templates = TemplateManager.getTemplates();
            const t = templates.find(t => t.id === overrideTemplateId);
            templateName = t ? t.name : 'Nieznany';
        } else {
            templateName = TemplateManager.getSelectedTemplateName();
        }

        // --- GHOST WRITER ---
        const settings = await StorageService.getSettings();
        const sentToGhost = GeminiBridge.sendToGeminiIframe(finalPrompt, settings.autoSubmit);

        // Backup: Kopiujemy do schowka
        let clipboardSuccess = false;
        try {
            await navigator.clipboard.writeText(finalPrompt);
            clipboardSuccess = true;
        } catch (clipErr) {
            console.warn("Błąd schowka (niekrytyczny):", clipErr);
        }
        
        const sourceInfo = contextData.isSelection ?
        'ZAZNACZENIE' : 'STRONĘ';
        
        if (sentToGhost) {
            const actionMsg = settings.autoSubmit ?
            "Wysłano do Gemini!" : "Wpisano do Gemini!";
            UIManager.showToast(`${actionMsg} (Tryb: ${templateName})`);
            UIManager.showButtonFeedback(btnId, true);
        } else if (clipboardSuccess) {
            UIManager.showToast(`Skopiowano ${sourceInfo} (Tryb: ${templateName})`);
            UIManager.showButtonFeedback(btnId, true);
        } else {
            throw new Error("Nie udało się ani wysłać do Gemini, ani skopiować do schowka.");
        }

    } catch (err) {
        console.error(err);
        UIManager.showToast("Błąd pobierania danych lub zapisu. Sprawdź konsolę.", true);
        UIManager.showButtonFeedback(btnId, false);
    }
}

async function checkPendingActions() {
    try {
        const data = await chrome.storage.local.get('pending_gemini_action');
        const action = data.pending_gemini_action;

        if (action && (Date.now() - action.timestamp < 30000)) {
            GeminiBridge.sendToGeminiIframe(action.text, action.autoSubmit);
            await chrome.storage.local.remove('pending_gemini_action');
        }
    } catch (err) {
        console.error("Błąd odczytu pending actions:", err);
    }
}

function getDraftTextFromIframe() {
    const iframe = document.getElementById('gemini-frame');
    if (!iframe || !iframe.contentWindow) {
        return Promise.resolve("");
    }

    return new Promise((resolve) => {
        const timeout = setTimeout(() => {
            window.removeEventListener('message', messageListener);
            resolve(null);
        }, 500);

        const messageListener = (event) => {
            if (event.data && event.data.action === "return_input_text") {
              
                clearTimeout(timeout);
                window.removeEventListener('message', messageListener);
                resolve(event.data.text);
            }
        };
        window.addEventListener('message', messageListener);

        iframe.contentWindow.postMessage({ action: "get_input_text" }, "*");
    });
}

async function handleDetach() {
    const iframe = document.getElementById('gemini-frame');
    const defaultUrl = chrome.runtime.getURL("sidepanel.html"); 
    const btnId = 'btn-detach';
    const btn = document.getElementById(btnId);
    const icon = btn.querySelector('i');
    const originalIconClass = 'fas fa-external-link-alt'; 
    icon.className = 'fas fa-circle-notch fa-spin';
    try {
        if (!iframe || !iframe.contentWindow) {
             await chrome.windows.create({
                 url: defaultUrl,
                 type: "popup", width: 500, height: 850, focused: true
             });
             icon.className = originalIconClass;
             UIManager.showButtonFeedback(btnId, true);
             return;
        }

        const receivedUrl = await new Promise((resolve, reject) => {
            const timeout = setTimeout(() => {
                window.removeEventListener('message', messageListener);
                resolve(null);
            }, 300);

            const messageListener = 
            (event) => {
                if (event.data && event.data.action === "return_current_url") {
                    clearTimeout(timeout);
                    window.removeEventListener('message', messageListener);
                    resolve(event.data.url);
               
                }
            };
            window.addEventListener('message', messageListener);
            iframe.contentWindow.postMessage({ action: "get_current_url" }, "*");
        });
        let newWindowUrl;
        
        if (receivedUrl && (receivedUrl.startsWith("https://gemini.google.com/") || receivedUrl.startsWith("https://duck.ai/"))) {
            newWindowUrl = defaultUrl + "?targetUrl=" + encodeURIComponent(receivedUrl);
        } else {
            newWindowUrl = defaultUrl;
            UIManager.showToast("Nie udało się pobrać aktywnego czatu. Otwieram stronę główną.", true);
        }

        await chrome.windows.create({
            url: newWindowUrl,
            type: "popup", width: 500, height: 850, focused: true
        });
        icon.className = originalIconClass;
        UIManager.showButtonFeedback(btnId, true);

    } catch (err) {
        console.error("Błąd otwierania okna lub komunikacji:", err);
        UIManager.showToast("Błąd otwierania okna.", true);
        icon.className = originalIconClass;
        UIManager.showButtonFeedback(btnId, false);
    }
}

const fs = require('fs');
const path = require('path');
const chokidar = require('chokidar');

// Domyślna konfiguracja (fallback)
let CONFIG = {
    watchFolder: './',
    outputFile: 'context_bundle.txt',
    allowedExtensions: ['.js', '.jsx', '.ts', '.tsx', '.css', '.html', '.json', '.py', '.md'],
    ignorePatterns: ['node_modules', '.git', 'dist', 'build', 'package-lock.json']
};

try {
    if (fs.existsSync('watcher.config.json')) {
        const userConfig = JSON.parse(fs.readFileSync('watcher.config.json', 'utf8'));
        CONFIG = { ...CONFIG, ...userConfig };
    }
    
    // ZABEZPIECZENIE: Zawsze dodaj plik wyjściowy do ignorowanych
    if (!CONFIG.ignorePatterns.includes(CONFIG.outputFile)) {
        CONFIG.ignorePatterns.push(CONFIG.outputFile);
    }
    
} catch (e) {
    console.warn('⚠️ Nie udało się wczytać configu, używam domyślnego.');
}

let debounceTimer;

const mergeFiles = () => {
    console.log('🔄 Wykryto zmiany. Generowanie kontekstu...');
    let fullContent = `--- PROJECT CONTEXT GENERATED AT ${new Date().toLocaleString()} ---\n\n`;
    
    fullContent += `// --- Configuration used: watcher.config.json ---\n`;
    fullContent += JSON.stringify(CONFIG, null, 2) + '\n\n';

    const walkSync = (dir, filelist = []) => {
        const files = fs.readdirSync(dir);
        files.forEach((file) => {
            const filePath = path.join(dir, file);
            const stat = fs.statSync(filePath);
            
            if (CONFIG.ignorePatterns.some(pattern => filePath.includes(pattern))) return;

            if (stat.isDirectory()) {
                walkSync(filePath, filelist);
            } else {
                if (CONFIG.allowedExtensions.includes(path.extname(file))) {
                    filelist.push(filePath);
                }
            }
        });
        return filelist;
    };

    try {
        const allFiles = walkSync(CONFIG.watchFolder);
        allFiles.forEach(filePath => {
            const content = fs.readFileSync(filePath, 'utf8');
            const relativePath = path.relative('.', filePath).replace(/\\/g, '/');
            fullContent += `\n// ==========================================\n// --- File: ${relativePath} ---\n// ==========================================\n\n${content}\n`;
        });

        fs.writeFileSync(CONFIG.outputFile, fullContent);
        console.log(`✅ Zaktualizowano: ${CONFIG.outputFile} (${allFiles.length} plików)`);
    } catch (err) {
        console.error('❌ Błąd:', err.message);
    }
};

const watcher = chokidar.watch(CONFIG.watchFolder, {
    ignored: (path) => CONFIG.ignorePatterns.some(s => path.includes(s)),
    persistent: true,
    ignoreInitial: true
});

const handleEvent = () => {
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(() => mergeFiles(), 300);
};

watcher.on('all', handleEvent);

console.log(`👀 Watcher uruchomiony. Obserwuję folder: ${CONFIG.watchFolder}`);
console.log(`📝 Plik wynikowy: ${CONFIG.outputFile}`);
console.log(`💡 Edytuj 'watcher.config.json' aby zmienić ustawienia.`);
mergeFiles();
